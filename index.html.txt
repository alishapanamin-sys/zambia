    // ========= SETTINGS =========

    // Countdown end time (Lusaka is typically UTC+2).
    // Change this date/time anytime you want.
    const END_AT = "2026-03-01T23:59:00+02:00";

    // Your CPA link (redirect target)
    const CPA_URL = "https://iconics.adsleads1.store/?utm_medium=45cb5b04ee2dd3d524876ef99d99f52ba4436745&utm_campaign=menhaza";

    // OPTIONAL: append captured fields to CPA URL as query params
    // (Some networks accept custom params; if not needed set to false)
    const APPEND_PARAMS_TO_CPA = true;

    // ========= COUNTDOWN =========
    const el = (id) => document.getElementById(id);
    const pad = (n) => String(n).padStart(2, "0");

    function tick(){
      const end = new Date(END_AT).getTime();
      const now = Date.now();
      let diff = Math.max(0, end - now);

      const d = Math.floor(diff / (1000*60*60*24));
      diff -= d * (1000*60*60*24);
      const h = Math.floor(diff / (1000*60*60));
      diff -= h * (1000*60*60);
      const m = Math.floor(diff / (1000*60));
      diff -= m * (1000*60);
      const s = Math.floor(diff / 1000);

      el("d").textContent = pad(d);
      el("h").textContent = pad(h);
      el("m").textContent = pad(m);
      el("s").textContent = pad(s);

      if (end - now <= 0) {
        el("urgencyText").textContent = "Registration closed";
      }
    }
    tick();
    setInterval(tick, 1000);
    el("year").textContent = new Date().getFullYear();

    // ========= FORM SUBMIT =========
    const form = document.getElementById("leadForm");
    const okMsg = document.getElementById("okMsg");
    const errMsg = document.getElementById("errMsg");

    function buildRedirectUrl(baseUrl, paramsObj){
      try{
        const u = new URL(baseUrl);
        Object.entries(paramsObj).forEach(([k,v]) => {
          if (v !== undefined && v !== null && String(v).trim() !== "") {
            u.searchParams.set(k, String(v).trim());
          }
        });
        return u.toString();
      }catch(e){
        // fallback if URL parsing fails
        const qs = Object.entries(paramsObj)
          .filter(([,v]) => v !== undefined && v !== null && String(v).trim() !== "")
          .map(([k,v]) => `${encodeURIComponent(k)}=${encodeURIComponent(String(v).trim())}`)
          .join("&");
        return baseUrl + (baseUrl.includes("?") ? "&" : "?") + qs;
      }
    }

    form.addEventListener("submit", (e) => {
      e.preventDefault();
      okMsg.style.display = "none";
      errMsg.style.display = "none";

      const name = document.getElementById("name").value.trim();
      const phone = document.getElementById("phone").value.trim();

      // Basic validation
      if (name.length < 3) {
        errMsg.textContent = "Please enter your full name.";
        errMsg.style.display = "block";
        return;
      }

      // Accept numbers, spaces, +, -, parentheses
      const phoneOk = /^[0-9+\-\s()]{8,}$/.test(phone);
      if (!phoneOk) {
        errMsg.textContent = "Please enter a valid mobile number.";
        errMsg.style.display = "block";
        return;
      }

      okMsg.style.display = "block";

      // Build redirect URL
      let goTo = CPA_URL;

      if (APPEND_PARAMS_TO_CPA) {
        // Rename keys if your CPA network expects specific parameter names
        goTo = buildRedirectUrl(CPA_URL, {
          full_name: name,
          mobile: phone,
          src: "landing_president_grant"
        });
      }

      // redirect after a short moment (better UX)
      setTimeout(() => {
        window.location.href = goTo;
      }, 900);
    });
